python Always show details Copy code import matplotlib.pyplot as plt import numpy as np # Sample data months = ['January', 'February', 'March', 'April'] products = ['Product A', 'Product B', 'Product C', 'Product D'] # Fake sales data (in units sold) sales_data = { 'Product A': [150, 200, 180, 220], 'Product B': [300, 330, 310, 400], 'Product C': [100, 120, 130, 150], 'Product D': [50, 80, 70, 90] } x = np.arange(len(months)) # the label locations width = 0.2 # the width of the bars fig, ax = plt.subplots(figsize=(10, 6)) # Plotting the data for each product for i, (product, sales) in enumerate(sales_data.items()): ax.bar(x + i * width, sales, width, label=product) # Labels and title ax.set_xlabel('Months') ax.set_ylabel('Units Sold') ax.set_title('Product Sales Over Months') ax.set_xticks(x + width * 1.5) ax.set_xticklabels(months) ax.legend() plt.tight_layout() plt.show() Product Sales Over Months
